def parse_sse_events(stream_body: str) -> list[dict]:
    """Split an SSE stream body into a list of parsed JSON event dicts."""
    events = []
    for chunk in stream_body.split("\n\n"):
        chunk = chunk.strip()
        if not chunk.startswith("data: "):
            continue
        try:
            events.append(json.loads(chunk[len("data: "):]))
        except json.JSONDecodeError:
            pass
    return events


def find_event_indices(events: list[dict], event_type: str) -> list[int]:
    return [i for i, e in enumerate(events) if e.get("type") == event_type]
