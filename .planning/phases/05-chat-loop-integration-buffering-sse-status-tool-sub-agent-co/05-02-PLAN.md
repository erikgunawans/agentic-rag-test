**Step 3 — Add the de-anon leaf transform helper.** This is a synchronous pure function (no I/O); the recursive walker awaits nothing in the de-anon direction. Add immediately after the constants block:

```python
def _deanon_leaf(s: str, registry: ConversationRegistry) -> str:
    """Pass-1 longest-surrogate-first registry transform for a single leaf string.

    Mirrors redaction_service.py:749-763 (Phase 4 D-71 Pass-1 algorithm).
    Pure registry lookup — never raises, never calls NER, never touches DB.
    """
    if _UUID_RE.fullmatch(s) or len(s) < _MIN_LEN:
        return s
    entries_sorted = sorted(
        registry.entries(),
        key=lambda m: (len(m.surrogate_value), len(m.real_value)),
        reverse=True,
    )
    out = s
    for m in entries_sorted:
        if not m.surrogate_value or not m.real_value:
            continue
        out = re.sub(
            re.escape(m.surrogate_value),
            m.real_value,
            out,
            flags=re.IGNORECASE,
        )
    return out
```

    via Pass-1 longest-surrogate-first transform. Pure lookup; no NER,

            needs only the registry).

    Returns:
        new dict (input never mutated) with surrogate strings replaced
        by real values; non-string leaves returned identity.
    """

def _deanon_walk(node: Any, registry: ConversationRegistry, depth: int) -> Any:
    """Synchronous recursive walker for the de-anon direction (no async I/O)."""
    if depth >= _MAX_DEPTH:
        return node
    if isinstance(node, dict):
        return {k: _deanon_walk(v, registry, depth + 1) for k, v in node.items()}
    if isinstance(node, list):
        return [_deanon_walk(v, registry, depth + 1) for v in node]
    if isinstance(node, tuple):
        return tuple(_deanon_walk(v, registry, depth + 1) for v in node)
    if isinstance(node, str):
        return _deanon_leaf(node, registry)
    return node

**Step 5 — Add the anon recursive walker (collect-then-batch leaf strategy).** Two-phase walk: phase 1 collects every transformable leaf string into a flat list with their paths; phase 2 calls `redact_text_batch` ONCE; phase 3 re-zips:

```python
@traced(name="redaction.anonymize_tool_output")
async def anonymize_tool_output(
    output: Any,
    registry: ConversationRegistry,
    redaction_service: RedactionService,
) -> Any:
    """Real-form tool output -> surrogate-form for LLM consumption.

    Collect-then-batch leaf strategy (D-91 + D-92): gather every
    transformable leaf string across the recursive structure, run ONE
    redact_text_batch call (single asyncio.Lock acquisition, single
    upsert batch), then re-zip the surrogate strings back into a NEW
    structure with the same shape as input.

    Args:
        output: arbitrary tool output (typically a dict from a tool's
            return value; e.g., search_documents returns
            {"results": [...]}, query_database returns {"rows": [...]}).
        registry: per-thread ConversationRegistry.
        redaction_service: required (the actual NER backend); calls
            redaction_service.redact_text_batch(collected, registry) once.

    Returns:
        new structure (input never mutated) with leaf strings replaced
        by their surrogate forms; non-string leaves returned identity.
        Skip rules: UUID-shaped strings + len<3 strings returned identity.
    """
    # Phase 1: collect transformable leaves with placeholder substitution.
    # We replace each transformable leaf with a sentinel index (an int) in
    # a SHADOW structure, building a parallel list of strings to batch-anon.
    leaves: list[str] = []

    def _collect(node: Any, depth: int) -> Any:
        if depth >= _MAX_DEPTH:
            return node
        if isinstance(node, dict):
            return {k: _collect(v, depth + 1) for k, v in node.items()}
        if isinstance(node, list):
            return [_collect(v, depth + 1) for v in node]
        if isinstance(node, tuple):
            return tuple(_collect(v, depth + 1) for v in node)
        if isinstance(node, str):
            if _UUID_RE.fullmatch(node) or len(node) < _MIN_LEN:
                return node  # identity — skip rule
            idx = len(leaves)
            leaves.append(node)
            return ("__PII_LEAF__", idx)  # marker tuple — re-zipped in phase 3
        return node

    shadow = _collect(output, depth=0)

    if not leaves:
        # Fast path — no transformable strings; return input shape verbatim.
        return output

    # Phase 2: ONE batched NER pass under ONE asyncio.Lock (D-92).
    anonymized = await redaction_service.redact_text_batch(leaves, registry)

    # Phase 3: re-zip — replace the marker tuples with anonymized strings.
    def _rezip(node: Any) -> Any:
        if isinstance(node, dict):
            return {k: _rezip(v) for k, v in node.items()}
        if isinstance(node, list):
            return [_rezip(v) for v in node]
        if isinstance(node, tuple):
            if (
                len(node) == 2
                and node[0] == "__PII_LEAF__"
                and isinstance(node[1], int)
            ):
                return anonymized[node[1]]
            return tuple(_rezip(v) for v in node)
        return node

    return _rezip(shadow)
```
