def _patched_settings(
    *,
    pii_redaction_enabled: bool = True,
    fuzzy_mode: str = "none",
    agents_enabled: bool = False,
) -> SimpleNamespace:
    """Override settings for a test — Phase 4 pattern extended with Phase 5 toggles."""
    real = get_settings()
    overrides = {f: getattr(real, f) for f in real.model_dump().keys()}
    overrides["pii_redaction_enabled"] = pii_redaction_enabled
    overrides["fuzzy_deanon_mode"] = fuzzy_mode
    overrides["agents_enabled"] = agents_enabled
    return SimpleNamespace(**overrides)


def _consume_sse(response) -> list[dict]:
    """Parse a TestClient SSE response into a list of event dicts.

    Each emitted line `data: {...}\\n\\n` becomes one dict in the result.
    Non-data lines (blank separators) are skipped.
    """
    events: list[dict] = []
    for raw_line in response.iter_lines():
        line = raw_line if isinstance(raw_line, str) else raw_line.decode("utf-8")
        if line.startswith("data: "):
            try:
                events.append(json.loads(line[6:]))
            except json.JSONDecodeError:
                continue
    return events


def _mock_llm_response(payload: dict) -> dict:
    """Wrap a payload as an AsyncOpenAI chat.completions.create response shape."""
    return {
        "choices": [{"message": {"content": json.dumps(payload), "role": "assistant"}}],
    }
