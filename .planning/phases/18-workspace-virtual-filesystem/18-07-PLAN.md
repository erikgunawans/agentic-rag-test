```ts
case 'workspace_updated': {
  const { file_path, operation, size_bytes, source } = event;
  setWorkspaceFiles((prev) => {
    const idx = prev.findIndex((f) => f.file_path === file_path);
    if (operation === 'delete') {
      return prev.filter((f) => f.file_path !== file_path);
    }
    const newEntry: WorkspaceFile = {
      file_path,
      size_bytes,
      source,
      mime_type: null, // server didn't send; will refresh on next list fetch
      updated_at: new Date().toISOString(),
    };
    if (idx === -1) {
      // create — prepend (most recent first)
      return [newEntry, ...prev];
    }
    // update — replace and move to top
    const next = [...prev];
    next.splice(idx, 1);
    return [newEntry, ...next];
  });
  break;
}
```

**Step B — Build `frontend/src/components/chat/WorkspacePanel.tsx`:**

Render file content as plain text inside a `<pre>` element using React's standard text-children rendering — DO NOT use any HTML-injection prop. React escapes string children, so `<pre>{content}</pre>` is safe by construction.

```tsx
import { useState } from 'react';
import {
  ChevronDown,
  ChevronRight,
  Download,
  FileText,
  File as FileIcon,
  Image as ImageIcon,
} from 'lucide-react';
import { Button } from '@/components/ui/button';
import { useI18n } from '@/i18n/I18nContext';
import { apiFetch } from '@/lib/api';
import type { WorkspaceFile } from '@/hooks/useChatState';

// Phase 18 — WS-07, WS-08, WS-11.
// Sidebar panel decoupled from Deep Mode — visible whenever the thread has any workspace files.
// Glass rule (CLAUDE.md / UI-SPEC): persistent panel — NO blurred backgrounds. Solid bg-card only.

interface WorkspacePanelProps {
  threadId: string;
  files: WorkspaceFile[];
}

function formatBytes(b: number): string {
  if (b < 1024) return `${b} B`;
  if (b < 1024 * 1024) return `${(b / 1024).toFixed(1)} KB`;
  return `${(b / (1024 * 1024)).toFixed(1)} MB`;
}

function isTextFile(mime: string | null): boolean {
  if (!mime) return true; // assume text when unknown (server defaults text)
  return mime.startsWith('text/') || mime === 'application/json';
}

function fileIconFor(mime: string | null) {
  if (mime?.startsWith('image/')) return ImageIcon;
  if (mime?.startsWith('text/')) return FileText;
  return FileIcon;
}

function relativeTime(iso: string): string {
  const diffMs = Date.now() - new Date(iso).getTime();
  const m = Math.floor(diffMs / 60000);
  if (m < 1) return 'just now';
  if (m < 60) return `${m}m ago`;
  const h = Math.floor(m / 60);
  if (h < 24) return `${h}h ago`;
  return new Date(iso).toLocaleDateString();
}

export function WorkspacePanel({ threadId, files }: WorkspacePanelProps) {
  const { t } = useI18n();
  const [collapsed, setCollapsed] = useState(false);
  const [expandedPath, setExpandedPath] = useState<string | null>(null);
  const [contentCache, setContentCache] = useState<Record<string, string>>({});

  if (files.length === 0) return null; // WS-11: hidden when empty

  const handleClick = async (f: WorkspaceFile) => {
    if (isTextFile(f.mime_type)) {
      if (expandedPath === f.file_path) {
        setExpandedPath(null);
        return;
      }
      if (!contentCache[f.file_path]) {
        const text = await apiFetch<string>(
          `/threads/${threadId}/files/${encodeURIComponent(f.file_path)}`,
          { responseType: 'text' },
        );
        setContentCache((c) => ({ ...c, [f.file_path]: text ?? '' }));
      }
      setExpandedPath(f.file_path);
    } else {
      // Binary: navigate to GET endpoint, browser follows 307 to signed URL → download
      window.open(`/threads/${threadId}/files/${encodeURIComponent(f.file_path)}`, '_blank');
    }
  };

  const SourceBadge = ({ source }: { source: WorkspaceFile['source'] }) => {
    const colorMap = {
      agent: 'bg-purple-500/20 text-purple-300',
      sandbox: 'bg-blue-500/20 text-blue-300',
      upload: 'bg-zinc-500/20 text-zinc-300',
    };
    return (
      <span className={`px-1.5 py-0.5 rounded text-xs ${colorMap[source]}`}>
        {t(`workspace.source.${source}`)}
      </span>
    );
  };

  return (
    <section
      className="bg-card border border-zinc-800 rounded-lg p-3 my-2"
      data-testid="workspace-panel"
    >
      <header className="flex items-center justify-between mb-2">
        <div className="flex items-center gap-2">
          <Button
            variant="ghost"
            size="sm"
            onClick={() => setCollapsed((c) => !c)}
            aria-label={collapsed ? t('workspace.expand') : t('workspace.collapse')}
          >
            {collapsed ? <ChevronRight size={14} /> : <ChevronDown size={14} />}
          </Button>
          <h3 className="text-sm font-medium">
            {t('workspace.title')} <span className="text-zinc-500">({files.length})</span>
          </h3>
        </div>
      </header>

      {!collapsed && (
        <ul className="space-y-1">
          {files.map((f) => {
            const Icon = fileIconFor(f.mime_type);
            const isExpanded = expandedPath === f.file_path;
            const text = isTextFile(f.mime_type);
            return (
              <li key={f.file_path}>
                <button
                  type="button"
                  onClick={() => handleClick(f)}
                  className="w-full flex items-center gap-2 px-2 py-1 rounded hover:bg-zinc-800/50 text-left"
                  data-testid={`workspace-file-${f.file_path}`}
                >
                  <Icon size={14} className="text-zinc-400" />
                  <span className="flex-1 text-sm truncate">{f.file_path}</span>
                  <span className="text-xs text-zinc-500">{formatBytes(f.size_bytes)}</span>
                  <SourceBadge source={f.source} />
                  <span className="text-xs text-zinc-600">{relativeTime(f.updated_at)}</span>
                  {!text && <Download size={12} className="text-zinc-400" />}
                </button>
                {isExpanded && text && (
                  <pre
                    className="mt-1 p-2 bg-zinc-900 rounded text-xs whitespace-pre-wrap max-h-96 overflow-auto"
                    data-testid={`workspace-content-${f.file_path}`}
                  >
                    {contentCache[f.file_path] ?? '...'}
                  </pre>
                )}
              </li>
            );
          })}
        </ul>
      )}
    </section>
  );
}
```

(Adjust the `apiFetch<string>` call to your actual API helper; if `responseType` isn't a thing, use `fetch` directly with `.text()`.)

## STRIDE Threat Register
| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-18-25 | T (Tampering) | XSS via file content with HTML-like tags | mitigate | content rendered via React text children only; no HTML-injection props used; React's default JSX behavior auto-escapes string children |
| T-18-26 | I (Information disclosure) | binary file path leaks via window.open referer | accept | signed URL is in the redirect target; the immediate `window.open` URL is the API path which contains thread_id (already authenticated context) |
