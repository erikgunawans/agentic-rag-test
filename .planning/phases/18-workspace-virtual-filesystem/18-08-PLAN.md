def _call_tool_via_chat(headers, thread_id, tool, args):
    """Helper that forces a chat-loop tool call. Implementation depends on whether
    the chat router has a 'force-tool' test hook or whether we need to do it via
    direct service-layer calls. Adapt to whatever the existing pattern is."""
    # Simplest: hit the workspace tool dispatch directly via a test-only endpoint,
    # OR use Python imports to call WorkspaceService directly with the user's token.
    from app.services.workspace_service import WorkspaceService
    import asyncio
    ws = WorkspaceService(token=headers["Authorization"].removeprefix("Bearer "))
    if tool == "write_file":
        return asyncio.run(ws.write_text_file(thread_id, args["file_path"], args["content"]))
    if tool == "read_file":
        return asyncio.run(ws.read_file(thread_id, args["file_path"]))
    if tool == "edit_file":
        return asyncio.run(ws.edit_file(thread_id, args["file_path"], args["old_string"], args["new_string"]))
    if tool == "list_files":
        return asyncio.run(ws.list_files(thread_id))
    raise ValueError(f"unknown tool {tool}")


def test_happy_path_lifecycle(fresh_thread):
    """Test 1: WS-01 + WS-02 — write, read, edit, list happy path."""
    tid, h = fresh_thread
    w = _call_tool_via_chat(h, tid, "write_file", {"file_path": "notes/x.md", "content": "hello"})
    assert w["ok"] and w["operation"] == "create" and w["size_bytes"] == 5

    r = _call_tool_via_chat(h, tid, "read_file", {"file_path": "notes/x.md"})
    assert r["ok"] and r["content"] == "hello"

    e = _call_tool_via_chat(h, tid, "edit_file", {"file_path": "notes/x.md", "old_string": "hello", "new_string": "hi"})
    assert e["ok"]

    r2 = _call_tool_via_chat(h, tid, "read_file", {"file_path": "notes/x.md"})
    assert r2["content"] == "hi"

    files = _call_tool_via_chat(h, tid, "list_files", {})
    assert any(f["file_path"] == "notes/x.md" for f in files)
