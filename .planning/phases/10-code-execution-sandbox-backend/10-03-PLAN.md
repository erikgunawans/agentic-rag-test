```python
"""SandboxService — Phase 10 / SANDBOX-01..06, 08.

Wraps llm-sandbox (Docker backend) with per-thread session persistence,
30-min TTL cleanup, per-call timeout, real-time stdout/stderr streaming,
and post-execution file upload to the sandbox-outputs Supabase bucket.

Decisions referenced:
  D-P10-01 — llm-sandbox library, Docker backend
  D-P10-04 — one container per thread, reused across calls
  D-P10-09 — sessions are ephemeral (in-memory dict only)
  D-P10-10 — 60s cleanup loop, 30-min idle TTL
  D-P10-11 — no per-user concurrent cap
  D-P10-12 — per-call execution timeout via asyncio.wait_for
  D-P10-13 — storage path {user_id}/{thread_id}/{execution_id}/{filename}
  D-P10-14 — signed URL TTL = 3600 seconds (1 hour)
  D-P10-08 — timeout/exception unify in stderr; exit_code distinguishes
"""

import asyncio
import logging
import os
import uuid
from dataclasses import dataclass
from datetime import datetime, timedelta
from functools import lru_cache
from typing import Awaitable, Callable

# llm-sandbox import path verified at execution time (see Plan 03 §interfaces note)
# Likely: `from llm_sandbox import SandboxSession`
# OR: `from llm_sandbox.docker import SandboxDockerSession as SandboxSession`
# Executor: run `pip show llm-sandbox` and read the package's __init__.py to confirm.

from app.config import get_settings
from app.database import get_supabase_client
from app.services.tracing_service import traced

logger = logging.getLogger(__name__)
settings = get_settings()


@dataclass
class SandboxSession:
    """Per-thread sandbox state. `container` is the opaque llm-sandbox handle."""
    container: object
    last_used: datetime
    thread_id: str


class SandboxService:
    """Singleton service managing one Docker container per thread."""

    def __init__(self) -> None:
        self._sessions: dict[str, SandboxSession] = {}
        self._cleanup_task: asyncio.Task | None = None
        self._lock = asyncio.Lock()

    # ── Session lifecycle ────────────────────────────────────────────

    async def _ensure_cleanup_task(self) -> None:
        """D-P10-10: lazy-spawn the 60s cleanup loop on first execute call."""
        if self._cleanup_task is None or self._cleanup_task.done():
            self._cleanup_task = asyncio.create_task(self._cleanup_loop())

    async def _cleanup_loop(self) -> None:
        """Close containers idle longer than 30 minutes. Runs forever; never raises."""
        while True:
            try:
                await asyncio.sleep(60)  # SANDBOX-02 spec: every 60s
                cutoff = datetime.utcnow() - timedelta(minutes=30)
                stale_ids = [
                    tid for tid, s in self._sessions.items() if s.last_used < cutoff
                ]
                for tid in stale_ids:
                    sess = self._sessions.pop(tid, None)
                    if sess is None:
                        continue
                    try:
                        # llm-sandbox lifecycle — exact close call verified at exec time
                        if hasattr(sess.container, "close"):
                            sess.container.close()
                        elif hasattr(sess.container, "__exit__"):
                            sess.container.__exit__(None, None, None)
                    except Exception as exc:
                        logger.warning(
                            "sandbox cleanup close failed thread=%s err=%s", tid, exc
                        )
            except asyncio.CancelledError:
                raise
            except Exception as exc:
                logger.error("sandbox cleanup loop error: %s", exc, exc_info=True)
                # Loop continues — never let cleanup take down the service.

    async def _get_or_create_session(self, thread_id: str) -> SandboxSession:
        """D-P10-04: one container per thread, lazy-create."""
        async with self._lock:
            sess = self._sessions.get(thread_id)
            if sess is not None:
                return sess
            # Create new container — exact llm-sandbox API verified at exec time.
            # Pseudo-code; executor adapts to real API:
            #     container = SandboxSession(
            #         image=settings.sandbox_image,
            #         backend="docker",
            #         keep_template=True,         # for variable persistence
            #         lang="python",
            #     )
            #     container.open()  # or `with container:` — depends on llm-sandbox API
            container = self._create_container()  # see helper below
            sess = SandboxSession(
                container=container,
                last_used=datetime.utcnow(),
                thread_id=thread_id,
            )
            self._sessions[thread_id] = sess
            return sess

    def _create_container(self) -> object:
        """Construct an llm-sandbox session bound to settings.sandbox_image.

        IMPORTANT: ensure DOCKER_HOST env is honored. llm-sandbox likely reads
        os.environ['DOCKER_HOST']; if not, pass it explicitly.
        """
        os.environ.setdefault("DOCKER_HOST", settings.sandbox_docker_host)
        # Executor: replace this stub with the real llm-sandbox API per pip-installed version.
        # Refer to llm-sandbox docs (Context7: try `mcp__context7__resolve-library-id` for "llm-sandbox").
        raise NotImplementedError(
            "Executor: replace with real llm-sandbox SandboxSession constructor"
        )

    # ── Execution ────────────────────────────────────────────────────

    @traced(name="sandbox_execute")
    async def execute(
        self,
        *,
        code: str,
        thread_id: str,
        user_id: str,
        stream_callback: Callable[[str, str], Awaitable[None]] | None = None,
    ) -> dict:
        """Run `code` in the thread's sandbox container.

        Returns the shape documented in the module docstring + Plan 03 §interfaces.
        """
        await self._ensure_cleanup_task()
        session = await self._get_or_create_session(thread_id)
        session.last_used = datetime.utcnow()

        execution_id = str(uuid.uuid4())
        started = datetime.utcnow()
        stdout_buf: list[str] = []
        stderr_buf: list[str] = []

        async def on_stdout(line: str) -> None:
            stdout_buf.append(line)
            if stream_callback:
                try:
                    await stream_callback("code_stdout", line)
                except Exception as exc:
                    logger.warning("stream_callback stdout error: %s", exc)

        async def on_stderr(line: str) -> None:
            stderr_buf.append(line)
            if stream_callback:
                try:
                    await stream_callback("code_stderr", line)
                except Exception as exc:
                    logger.warning("stream_callback stderr error: %s", exc)

        exit_code = 0
        error_type: str | None = None
        try:
            # D-P10-12: per-call timeout via asyncio.wait_for.
            # llm-sandbox API for streaming output may differ; executor adapts:
            # one common shape:
            #     result = await asyncio.wait_for(
            #         session.container.run(code, on_stdout=on_stdout, on_stderr=on_stderr),
            #         timeout=settings.sandbox_max_exec_seconds,
            #     )
            #     exit_code = result.exit_code
            # If the library's run() is synchronous, wrap in run_in_executor.
            result = await asyncio.wait_for(
                self._run_in_session(session, code, on_stdout, on_stderr),
                timeout=settings.sandbox_max_exec_seconds,
            )
            exit_code = getattr(result, "exit_code", 0)
        except asyncio.TimeoutError:
            await on_stderr(
                f"Execution timed out after {settings.sandbox_max_exec_seconds}s"
            )
            exit_code = -1
            error_type = "timeout"
        except Exception as exc:
            # D-P10-08: tracebacks unify in stderr
            await on_stderr(f"Execution failed: {exc}")
            exit_code = 1
            error_type = "exception"

        execution_ms = int(
            (datetime.utcnow() - started).total_seconds() * 1000
        )

        # D-P10-13/14: collect generated files, upload, sign URLs
        files = await self._collect_and_upload_files(
            session=session,
            user_id=user_id,
            thread_id=thread_id,
            execution_id=execution_id,
        )

        return {
            "stdout": "".join(stdout_buf),
            "stderr": "".join(stderr_buf),
            "exit_code": exit_code,
            "error_type": error_type,
            "execution_ms": execution_ms,
            "files": files,
            "execution_id": execution_id,
        }

    async def _run_in_session(
        self,
        session: SandboxSession,
        code: str,
        on_stdout: Callable[[str], Awaitable[None]],
        on_stderr: Callable[[str], Awaitable[None]],
    ):
        """Bridge to llm-sandbox's run API. Executor adapts to actual signature."""
        raise NotImplementedError(
            "Executor: bridge to llm-sandbox run() — sync vs async, callback vs poll"
        )

    # ── File upload ──────────────────────────────────────────────────

    async def _collect_and_upload_files(
        self,
        *,
        session: SandboxSession,
        user_id: str,
        thread_id: str,
        execution_id: str,
    ) -> list[dict]:
        """List files in /sandbox/output/, upload each, return signed-URL list.

        Returns: list of {filename, size_bytes, signed_url, storage_path}.

        Path scheme (D-P10-13): {user_id}/{thread_id}/{execution_id}/{filename}.
        Signed URL TTL (D-P10-14): 3600 seconds.
        """
        try:
            file_list = self._list_output_files(session)  # llm-sandbox-specific
        except Exception as exc:
            logger.warning("sandbox list output files failed: %s", exc)
            return []

        if not file_list:
            return []

        client = get_supabase_client()
        uploaded: list[dict] = []
        for filename, content in file_list:
            storage_path = f"{user_id}/{thread_id}/{execution_id}/{filename}"
            try:
                client.storage.from_("sandbox-outputs").upload(
                    storage_path,
                    content,
                    {"content-type": "application/octet-stream"},
                )
                signed = client.storage.from_("sandbox-outputs").create_signed_url(
                    storage_path, 3600  # D-P10-14: 1-hour TTL
                )
                signed_url = signed.get("signedURL") or signed.get("signed_url") or ""
                uploaded.append({
                    "filename": filename,
                    "size_bytes": len(content),
                    "signed_url": signed_url,
                    "storage_path": storage_path,
                })
            except Exception as exc:
                logger.warning(
                    "sandbox upload failed file=%s err=%s", filename, exc
                )
        return uploaded

    def _list_output_files(self, session: SandboxSession) -> list[tuple[str, bytes]]:
        """Return [(filename, content_bytes)] from /sandbox/output/ inside the container.

        Executor: implement using llm-sandbox's exec / copy API. Most libraries provide
        either a `container.copy_from(src, dst)` or `container.exec_run("ls /sandbox/output/")`.
        """
        raise NotImplementedError(
            "Executor: implement output-file listing using llm-sandbox exec/copy API"
        )


@lru_cache
def get_sandbox_service() -> SandboxService:
    """Process-singleton accessor."""
    return SandboxService()
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| LLM-generated code → Docker container | TRUE isolation per D-P10-01 — code runs in separate container, cannot access backend memory or env vars |
| Backend service → Docker socket (`/var/run/docker.sock`) | Backend has root-equivalent control over the host's Docker daemon. Container escape from sandbox would expose this |
| sandbox container → Supabase secrets | NONE — backend uploads files on behalf of user via service-role client; sandbox container has no Supabase credentials |
| User-controlled `code` string → execution | Untrusted input executed as Python — full sandbox isolation is the only mitigation |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-12 | Elevation of Privilege | LLM code escapes Docker → backend host | mitigate | llm-sandbox isolation (D-P10-01); SandboxDockerfile USER 1000:1000 (Plan 02 T-10-07); per-call timeout (D-P10-12) limits exploit window; no privileged mode |
| T-10-13 | Information Disclosure | Backend env vars leak via `os.environ` inside sandbox | mitigate | llm-sandbox creates separate container; backend's env not inherited. Verified by D-P10-01 architectural choice |
| T-10-14 | Denial of Service | Infinite-loop / fork-bomb code | mitigate | `asyncio.wait_for(..., timeout=settings.sandbox_max_exec_seconds)` — 30s default. Container killed on timeout; session reused on next call |
| T-10-15 | Tampering | Path traversal in filename: code writes `/sandbox/output/../../etc/passwd` | mitigate | sandbox container is isolated — even if `..` traversal succeeds inside container, it cannot escape Docker namespace. Storage upload path is server-constructed (`{user_id}/{thread_id}/{execution_id}/{filename}`) — filename comes from `os.listdir()` not user input |
| T-10-16 | Spoofing | Wrong thread_id passed → user A reads user B's session | mitigate | Plan 04 (`tool_service._execute_code`) sources thread_id from `tool_context["thread_id"]` which Plan 05 (chat.py) populates from the validated `body.thread_id` (chat.py line 99 thread-ownership check) |
| T-10-17 | Information Disclosure | Signed URL leaks via SSE stream / persisted message | mitigate | 1-hour TTL (D-P10-14) bounds exposure; URL refreshed on read in Plan 06's list endpoint. URL contains no PII directly |
| T-10-18 | Denial of Service | Memory leak from never-closed sessions | mitigate | 30-min idle cleanup (D-P10-10) + container.close() in cleanup loop. Verified by Test 6 in Task 1 |
| T-10-19 | Spoofing | Filename in storage upload contains `/` (causes 5+ segment path) | mitigate | `_list_output_files` returns flat filenames from `os.listdir("/sandbox/output/")` (no recursion). Documentation in code comment that nested paths are NOT supported in v1.1 |
