```python
"""Code Execution router — Phase 10 / D-P10-17.

Read-only endpoint that exposes the immutable `code_executions` audit log to
authenticated users. RLS auto-filters to own rows; super_admin sees all
(D-P10-15). Signed URLs in the `files` JSONB are refreshed at read time
because their TTL is 1 hour (D-P10-14) and rows can be days old.

Used by Phase 11's Code Execution Panel UI to render execution history.
"""

from datetime import datetime
import logging

from fastapi import APIRouter, Depends, Query
from pydantic import BaseModel, ConfigDict

from app.dependencies import get_current_user
from app.database import get_supabase_authed_client, get_supabase_client

logger = logging.getLogger(__name__)

router = APIRouter(prefix="/code-executions", tags=["code_executions"])


class CodeExecutionResponse(BaseModel):
    """Mirrors public.code_executions row shape (migration 036)."""
    id: str
    user_id: str
    thread_id: str
    code: str
    description: str | None = None
    stdout: str
    stderr: str
    exit_code: int
    execution_ms: int
    status: str  # 'success' | 'error' | 'timeout'
    files: list[dict]  # [{filename, size_bytes, signed_url, storage_path}]
    created_at: datetime

    model_config = ConfigDict(from_attributes=True)


def _refresh_signed_urls(files: list[dict]) -> list[dict]:
    """D-P10-14: regenerate signed URLs at read time (1-hour TTL).

    Iterates files[]; for each entry with a `storage_path`, replaces `signed_url`
    with a freshly-signed URL. Service-role client used because storage signed-URL
    generation requires admin privilege; SELECT RLS already gated row access.

    Failures are logged but do NOT block the response — stale URL is preferable
    to no response. Frontend can re-request if a URL 403s.
    """
    if not files:
        return []
    svc = get_supabase_client()
    refreshed: list[dict] = []
    for f in files:
        new_f = dict(f)  # don't mutate the caller's dict
        path = f.get("storage_path")
        if path:
            try:
                signed = svc.storage.from_("sandbox-outputs").create_signed_url(
                    path, 3600
                )
                # supabase-py 2.7.4 returns dict with key 'signedURL' (JS-style); fall back
                new_f["signed_url"] = (
                    signed.get("signedURL") or signed.get("signed_url") or new_f.get("signed_url", "")
                )
            except Exception as exc:
                logger.warning(
                    "code_executions signed URL refresh failed path=%s err=%s",
                    path, exc,
                )
                # keep stale URL — don't drop the file entry
        refreshed.append(new_f)
    return refreshed


@router.get("", response_model=dict)
async def list_code_executions(
    thread_id: str = Query(..., description="UUID of the thread to fetch executions for"),
    limit: int = Query(50, ge=1, le=100),
    offset: int = Query(0, ge=0),
    user: dict = Depends(get_current_user),
) -> dict:
    """List code executions for a thread.

    D-P10-15 RLS: own rows always visible; super_admin sees all rows.
    D-P10-17: this endpoint is the data source for Phase 11's Code Execution Panel.

    Response envelope (matches Phase 7 skills.py:318):
        {"data": [CodeExecutionResponse, ...], "count": int}
    """
    client = get_supabase_authed_client(user["token"])
    try:
        result = (
            client.table("code_executions")
            .select("*")
            .eq("thread_id", thread_id)
            .order("created_at", desc=True)
            .range(offset, offset + limit - 1)
            .execute()
        )
    except Exception as exc:
        logger.error("code_executions list query failed: %s", exc)
        return {"data": [], "count": 0}

    rows = result.data or []
    enriched: list[dict] = []
    for row in rows:
        # D-P10-14: refresh signed URLs at read time
        row["files"] = _refresh_signed_urls(row.get("files") or [])
        enriched.append(CodeExecutionResponse(**row).model_dump(mode="json"))
    return {"data": enriched, "count": len(enriched)}
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User JWT → endpoint | `get_current_user` validates the JWT and resolves `user_profiles.is_active` |
| User token → RLS-scoped query | `get_supabase_authed_client(token)` ensures RLS policies apply |
| Service-role → `create_signed_url` | Service-role bypasses RLS for storage-API access (signed URL generation requires it) |
| `thread_id` query param → eq filter | User-controlled string; even if malicious user passes another user's thread_id, RLS blocks the result |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-31 | Information Disclosure | User passes another user's thread_id and reads their executions | mitigate | RLS policy `code_executions_select` (Plan 01) auto-filters by `user_id = auth.uid()`. The `.eq("thread_id", thread_id)` is a NARROWING filter, not an authorization gate. Verified by Test 5 |
| T-10-32 | Information Disclosure | Signed URL refresh exposes another user's storage path | mitigate | We only refresh URLs for files in rows the user can SELECT (RLS-gated). `storage_path` from row already starts with the user's own UID per D-P10-13 |
| T-10-33 | Spoofing | super_admin claim forged by user via JWT manipulation | accept | Supabase JWT is signed; `app_metadata.role` is server-controlled. JWT signature verification handled upstream by `get_current_user`. Forgery requires the signing key |
| T-10-34 | Denial of Service | User passes huge limit | mitigate | `Query(50, ge=1, le=100)` enforces upper bound. FastAPI rejects out-of-range with 422 |
| T-10-35 | Tampering | User edits stale signed_url to access another user's file | accept | The OLD signed URL was issued for the specific user's path; even if leaked, the URL itself encodes the path. Cross-user access via stolen URL is bounded by URL TTL (1 hour). Future: add IP-binding if needed |
| T-10-36 | Repudiation | List endpoint is read-only — no audit emit | accept | Read endpoints don't typically trigger audit per existing pattern (skills.py list_skills line 278 has no log_action). Acceptable — code_executions itself IS the audit log |
| T-10-37 | Information Disclosure | Logged exception in `_refresh_signed_urls` exposes storage_path | mitigate | Log message includes path but goes only to backend logs (not response body). path is already user-scoped, so logging it for own user is acceptable |
</threat_model>
