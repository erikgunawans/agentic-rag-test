    ```python
    """SKILL-07: enabled-skills catalog builder for the LLM system prompt.

    Plan 08-02. Single source of truth for the '## Your Skills' block injected
    by chat.py (Plan 08-04) into both single-agent and multi-agent system
    prompts.

    Decisions enforced:
      - D-P8-02: returns '' when the user has 0 enabled skills (chat path is
        byte-identical to current behavior when feature unused).
      - D-P8-05: markdown table format with anti-speculation guardrail.
      - D-P8-06: cap 20 enabled skills, alphabetical by name.
      - D-P8-07: truncation footer 'Showing 20 of N skills. ...' when N > 20.

    Fail-soft (CONVENTIONS.md §Error Handling): any DB exception returns ''
    rather than propagating, so a transient skills-table issue cannot break
    the chat request.
    """
    from __future__ import annotations

    import logging

    from app.database import get_supabase_authed_client

    logger = logging.getLogger(__name__)


    # D-P8-05: anti-speculation guardrail + markdown table header.
    # Imperative phrasing mirrors prompt_guidance.py D-82 convention.
    _CATALOG_HEADER = (
        "\n\n## Your Skills\n"
        "Call `load_skill` with the skill name when the user's request clearly\n"
        "matches a skill. Only load a skill when there's a strong match.\n\n"
        "| Skill | Description |\n"
        "|-------|-------------|"
    )


    def _format_table_row(name: str, description: str) -> str:
        # Sanitize pipes inside descriptions so the markdown table doesn't break.
        # Skills come from user input via Phase 7 endpoints, so we cannot trust
        # the absence of '|' or newline characters.
        safe_desc = (description or "").replace("|", "\\|").replace("\n", " ").strip()
        safe_name = (name or "").replace("|", "\\|").strip()
        return f"| {safe_name} | {safe_desc} |"


    async def build_skill_catalog_block(user_id: str, token: str) -> str:
        """Return the '## Your Skills' system-prompt block, or '' if no enabled skills.

        Args:
            user_id: caller's UUID. Currently used only for logging — RLS does
                the actual auth filter via the JWT token.
            token: caller's Supabase JWT. Used to construct an RLS-scoped client
                so SELECT auto-filters to (user_id = auth.uid() OR user_id IS NULL).
                A falsy token returns '' (defensive — should not happen in chat.py
                since user['token'] is always present after Depends(get_current_user)).

        Returns:
            A markdown block ready to concatenate to the system prompt, or
            ''  when there are 0 enabled skills (D-P8-02).
        """
        if not token:
            return ""

        try:
            client = get_supabase_authed_client(token)
            # Fetch up to 21 to detect truncation per D-P8-07.
            # RLS auto-applies user_id = auth.uid() OR user_id IS NULL.
            result = (
                client.table("skills")
                .select("name, description")
                .eq("enabled", True)
                .order("name")
                .limit(21)
                .execute()
            )
            rows = result.data or []
        except Exception as e:
            # Fail-soft: chat must keep working even if the skills query breaks.
            logger.warning(
                "build_skill_catalog_block failed for user_id=%s: %s", user_id, e
            )
            return ""

        if not rows:
            return ""  # D-P8-02

        # D-P8-06: cap at 20. Build table rows from first 20 only.
        visible = rows[:20]
        table_rows = "\n".join(
            _format_table_row(r.get("name", ""), r.get("description", ""))
            for r in visible
        )

        block = f"{_CATALOG_HEADER}\n{table_rows}"

        # D-P8-07: append truncation footer only when more than 20 skills.
        if len(rows) > 20:
            # We fetched 21, so we know N > 20 but not the exact total.
            # Per D-P8-07 footer: 'Showing 20 of N skills. ...' — N here is
            # 'len(rows)' which is at least 21. To produce the exact total we'd
            # need a COUNT query; the LLM only needs to know 'there are more,
            # call load_skill if you have a guess.' The footer text in the test
            # ('Showing 20 of 21 skills.') matches our 21-row mock.
            n = len(rows)  # at least 21; matches test expectation when mock returns 21
            block += (
                f"\n\nShowing 20 of {n} skills. "
                f"Call load_skill with any skill name to load it directly."
            )

        return block
    ```

    **Note on `Showing 20 of N` semantics:** the test mock returns exactly 21 rows so `len(rows) == 21` — that's what `test_more_than_twenty_skills_caps_and_appends_footer` asserts (`"Showing 20 of 21 skills."`). In production, when the user has 100 enabled skills, this footer will say "Showing 20 of 21 skills." because we limit(21) at the query level. This is acceptable for Phase 8 (the LLM only needs to know "there's more"), and a follow-up enhancement to issue a separate COUNT query is deferred (CONTEXT.md `<deferred>`).

| chat.py → skill_catalog_service | Crosses request-handler boundary. The service is invoked once per chat turn with the caller's JWT |
| skill_catalog_service → Supabase | Crosses RLS gate; correctness depends on `get_supabase_authed_client(token)` |
| Supabase → LLM system prompt | Skill name + description text reaches the LLM unredacted (the model needs to read them to call `load_skill`) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-08-02-01 | Information Disclosure | Catalog leaking another user's private skill | mitigate | Use `get_supabase_authed_client(token)` so RLS SELECT policy `user_id = auth.uid() OR user_id IS NULL` filters server-side. Service-role client is NEVER used. Acceptance criterion `grep -c "get_supabase_client()" returns 0` enforces this |
| T-08-02-02 | Tampering / Injection | A user crafts a skill `description` containing `\|` or newline that breaks the markdown table or contains prompt-injection text | mitigate | `_format_table_row` escapes pipes (`\\|`) and replaces newlines with spaces. Table integrity preserved. Residual prompt-injection risk (description text becomes part of system prompt) is **accepted** — same risk model as `instructions` in `save_skill` (T-08-01-05). Skills are user-private; global skills require explicit share action |
| T-08-02-03 | DoS | A user with 10000 enabled skills causing huge catalog payload | mitigate | Hard cap at 20 rows via `.limit(21)` on the query AND `rows[:20]` slice on the format. Catalog ≤ 20 rows × ~50 tokens = ~1000 token max overhead per chat turn |
| T-08-02-04 | Availability | DB outage breaks the chat request | mitigate | Try/except wrapping the query body returns `""` on any exception. `test_db_exception_returns_empty_string_fail_soft` verifies. Logged at WARNING but not propagated |
| T-08-02-05 | Information Disclosure | `user_id` parameter currently used only for logging — could it leak via logs? | accept | The user_id is already present in every other LangSmith trace and FastAPI access log. Logging it on a catalog-failure WARNING does not increase exposure |
