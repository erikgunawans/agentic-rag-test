**Change C — Add file handlers** (after the existing `handleExport` from Plan 03):

```tsx
function fileIconFor(mime: string | null) {
  if (mime && mime.startsWith('text/')) return FileText
  if (mime && mime.startsWith('application/') && /(json|xml|yaml|javascript|typescript|markdown)/.test(mime)) return FileCode
  return File
}

function formatFileSize(bytes: number): string {
  if (bytes < 1024) return `${bytes} B`
  if (bytes < 1024 * 1024) return `${Math.round(bytes / 1024)} KB`
  return `${(bytes / (1024 * 1024)).toFixed(1)} MB`
}

async function handleUpload(e: React.ChangeEvent<HTMLInputElement>) {
  const file = e.target.files?.[0]
  e.target.value = ''  // allow re-upload of same filename later
  if (!file || !selectedSkill) return
  if (file.size > 10 * 1024 * 1024) {
    setErrorBanner(t('skills.errorFileSize'))
    return
  }
  setUploading(true)
  setErrorBanner(null)
  try {
    const formData = new FormData()
    formData.append('file', file)
    await apiFetch(`/skills/${selectedSkill.id}/files`, { method: 'POST', body: formData })
    // Re-fetch file list
    const filesRes = await apiFetch(`/skills/${selectedSkill.id}/files`)
    const filesBody = await filesRes.json()
    setSkillFiles(filesBody.data || [])
  } catch {
    setErrorBanner(t('skills.errorUpload'))
  } finally {
    setUploading(false)
  }
}

async function handleDeleteFile(fileId: string, filename: string) {
  if (!selectedSkill) return
  if (!confirm(t('skills.fileDeleteConfirm'))) return
  try {
    await apiFetch(`/skills/${selectedSkill.id}/files/${fileId}`, { method: 'DELETE' })
    // Re-fetch file list
    const filesRes = await apiFetch(`/skills/${selectedSkill.id}/files`)
    const filesBody = await filesRes.json()
    setSkillFiles(filesBody.data || [])
    // Close preview if it was showing this file
    if (previewFile?.fileId === fileId) setPreviewFile(null)
  } catch {
    setErrorBanner(t('skills.errorUpload'))  // generic per UI-SPEC; reuse upload error string
  }
}

async function openPreview(file: SkillFile) {
  if (!selectedSkill) return
  setPreviewLoading(true)
  setPreviewFile({
    fileId: file.id,
    filename: file.filename,
    mimeType: file.mime_type,
    content: null,
    truncated: false,
    totalBytes: file.size_bytes,
  })
  try {
    const res = await apiFetch(`/skills/${selectedSkill.id}/files/${file.id}/content`)
    const body = await res.json()
    // Backend response shape (per D-P8-12 / D-P8-13):
    //   text:   { filename, content, truncated, total_bytes, mime_type }
    //   binary: { filename, mime_type, size_bytes, readable: false, message }
    if (body.readable === false || typeof body.content !== 'string') {
      setPreviewFile({
        fileId: file.id,
        filename: body.filename ?? file.filename,
        mimeType: body.mime_type ?? file.mime_type,
        content: null,
        truncated: false,
        totalBytes: body.size_bytes ?? file.size_bytes,
      })
    } else {
      setPreviewFile({
        fileId: file.id,
        filename: body.filename ?? file.filename,
        mimeType: body.mime_type ?? file.mime_type,
        content: body.content,
        truncated: Boolean(body.truncated),
        totalBytes: body.total_bytes ?? file.size_bytes,
      })
    }
  } catch {
    setErrorBanner(t('skills.errorUpload'))
    setPreviewFile(null)
  } finally {
    setPreviewLoading(false)
  }
}

function closePreview() {
  setPreviewFile(null)
  setCopyState('idle')
}

async function handleCopy() {
  if (!previewFile?.content) return
  try {
    await navigator.clipboard.writeText(previewFile.content)
    setCopyState('copied')
    setTimeout(() => setCopyState('idle'), 1500)
  } catch {
    // clipboard API can fail on http (non-localhost); silently no-op.
  }
}

function handleDownloadPreview() {
  if (!previewFile) return
  if (previewFile.content !== null) {
    // Text — download from in-memory content as blob
    const blob = new Blob([previewFile.content], { type: previewFile.mimeType ?? 'text/plain' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a')
    a.href = url
    a.download = previewFile.filename
    document.body.appendChild(a)
    a.click()
    document.body.removeChild(a)
    URL.revokeObjectURL(url)
  } else {
    // Binary — re-fetch raw bytes via the content endpoint with an Accept header that asks for the raw stream.
    // The Phase 8 endpoint (per backend/app/routers/skills.py — verify field) returns metadata-only for binary.
    // Fallback: rely on browser-direct fetch with auth header via apiFetch(.../content?raw=1). If unsupported,
    // show error banner and abort. Verify the backend behavior at execute time.
    if (!selectedSkill) return
    apiFetch(`/skills/${selectedSkill.id}/files/${previewFile.fileId}/content?raw=1`)
      .then(r => r.blob())
      .then(blob => {
        const url = URL.createObjectURL(blob)
        const a = document.createElement('a')
        a.href = url
        a.download = previewFile.filename
        document.body.appendChild(a)
        a.click()
        document.body.removeChild(a)
        URL.revokeObjectURL(url)
      })
      .catch(() => setErrorBanner(t('skills.errorUpload')))
  }
}

// ---- Import flow ----
async function handleImportZip(e: React.ChangeEvent<HTMLInputElement>) {
  const file = e.target.files?.[0]
  e.target.value = ''
  if (!file) return
  setImportInProgress(true)
  setErrorBanner(null)
  try {
    const formData = new FormData()
    formData.append('file', file)
    const res = await apiFetch('/skills/import', { method: 'POST', body: formData })
    const body = (await res.json()) as ImportSummary
    setImportSummary(body)
    await fetchSkills()
  } catch {
    setErrorBanner(t('skills.errorImport'))
  } finally {
    setImportInProgress(false)
  }
}
```

```tsx
    </div>
```

| client to /skills/{id}/files POST | User-supplied file bytes cross to backend storage (Supabase). Backend enforces size + RLS. |
| /skills/{id}/files/{file_id}/content response to <pre> render | Text content (potentially user-uploaded malicious content) is rendered into a `<pre>` block. |

| /skills/import response to import summary modal | Per-skill error strings come from backend validation messages — user-visible. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-09-04-01 | Tampering | client-side 10 MB cap on upload | mitigate | The frontend pre-check (`if (file.size > 10 * 1024 * 1024) return error`) is UX optimization. Backend enforces the cap authoritatively at `POST /skills/{id}/files` (D-P7-08). A user bypassing the client check would still get a 4xx from the backend; the frontend should also surface that error via `errorBanner`. The current code's `catch` block already handles this. |
| T-09-04-02 | XSS via <pre> rendering of user-uploaded text content | preview drawer | mitigate | The `<pre>` block uses JSX `{previewFile.content}` — React renders the string as a text node, escaping HTML by default. There is no raw-HTML render path for file content. This is a true defense-in-depth: even if a malicious actor uploads a file containing `<script>alert(1)</script>`, it renders as visible literal text. CONFIRM: the executor must NOT change `<pre>{...}</pre>` to use any raw-HTML rendering primitive. |
| T-09-04-03 | XSS via filename rendered into aria-labelledby header / title attribute | preview drawer | mitigate | Filenames come from the backend `skill_files.filename` column, which Phase 7 sanitizes during ZIP import (storage path is `{owner_id}/{skill_id}/{filename}` — names with path separators or null bytes are rejected by the parser). The header renders the filename via JSX text content (`<h2>{previewFile.filename}</h2>`), so it's auto-escaped. Title attribute use is also safe (browsers do not interpret HTML in `title`). |
| T-09-04-04 | Information disclosure | clipboard write | accept | The Copy button only writes the content already visible to the user (the same content rendered in `<pre>`). The user already has access to it via the GET request. No additional disclosure risk. |
| T-09-04-05 | EoP | DELETE /skills/{id}/files/{file_id} | mitigate | UI shows the trash icon only when `isOwnPrivate` (own private skill). Frontend hiding is UX; backend is authoritative. |
| T-09-04-06 | Tampering | import ZIP processing | accept | Backend (Phase 7 `skill_zip_service`) parses and validates the ZIP, rejects malformed entries, enforces 10 MB per-file + 50 MB total caps. | T-09-04-08 | Open redirect / link injection via filename in download anchor | preview download | mitigate | The download anchor uses `a.href = URL.createObjectURL(blob)` and `a.download = previewFile.filename`. The `download` attribute is a plain string — browsers do not interpret it as URL or HTML. Even a filename like `<script>` would be rendered as a plain filename in the OS save dialog. The href comes from `URL.createObjectURL`, which produces a same-origin blob: URL — no external redirect possible. |
